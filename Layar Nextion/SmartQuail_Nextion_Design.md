### ESP32 Code untuk Update Nextion:
```cpp
#include <HardwareSerial.h>

HardwareSerial NextionSerial(2); // Use UART2

void setup() {
  NextionSerial.begin(9600, SERIAL_8N1, 17, 16); // RX=17, TX=16
}

// Update temperature display
void updateTemperature(float temp) {
  String cmd = "tTemp.txt=\"" + String(temp, 1) + "Â°C\"";
  sendToNextion(cmd);
}

// Update humidity display
void updateHumidity(int humid) {
  String cmd = "tHumid.txt=\"" + String(humid) + "%\"";
  sendToNextion(cmd);
}

// Update THI display and progress bar
void updateTHI(float thi) {
  String cmd = "tTHI.txt=\"" + String(thi, 1) + "\"";
  sendToNextion(cmd);
  
  // Update progress bar (scale 60-100 to 0-100)
  int progress = map(thi, 60, 100, 0, 100);
  String cmdBar = "jTHI.val=" + String(progress);
  sendToNextion(cmdBar);
  
  // Update status color
  if (thi < 72) {
    sendToNextion("tTHIStatus.txt=\"NORMAL\"");
    sendToNextion("tTHIStatus.pco=5765"); // Green
  } else if (thi < 78) {
    sendToNextion("tTHIStatus.txt=\"WARNING\"");
    sendToNextion("tTHIStatus.pco=64480"); // Orange
  } else {
    sendToNextion("tTHIStatus.txt=\"DANGER\"");
    sendToNextion("tTHIStatus.pco=63488"); // Red
  }
}

// Send command to Nextion
void sendToNextion(String cmd) {
  NextionSerial.print(cmd);
  NextionSerial.write(0xFF);
  NextionSerial.write(0xFF);
  NextionSerial.write(0xFF);
}

// Receive command from Nextion (button press, etc)
void checkNextionInput() {
  if (NextionSerial.available()) {
    String received = NextionSerial.readStringUntil('\n');
    
    if (received.startsWith("MIST:")) {
      int state = received.substring(5).toInt();
      digitalWrite(RELAY_MIST, state);
    }
    else if (received.startsWith("FAN:")) {
      int state = received.substring(4).toInt();
      digitalWrite(RELAY_FAN, state);
    }
    else if (received.startsWith("FEED:")) {
      activateFeeder();
    }
  }
}
```
